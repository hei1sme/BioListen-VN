class BioListenModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.backbone = models.efficientnet_v2_s(pretrained=True)
        # Custom Twin Heads
        self.species_head = nn.Linear(1280, 5)
        self.threat_head = nn.Linear(1280, 3)

    def forward(self, x):
        features = self.backbone.features(x)
        x = F.adaptive_avg_pool2d(features, 1).flatten(1)
        return self.species_head(x), self.threat_head(x)
